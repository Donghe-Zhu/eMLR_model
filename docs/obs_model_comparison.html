1 Read joined GLODAP + model data

# set path to model analysis/subsetting output files
path_model_analysis <-
  "/nfs/kryo/work/loher/CESM_output/RECCAP2/analysis/"

# read file for GLODAP observations + subsetted model variables
GLODAP_model <- 
  read_csv(paste(
    paste(path_model_analysis, "glodapv2_2020_synthetic_data/", sep = ""),
    "GLODAP_with_model_subset.csv",
    sep = ""
  ))

2 Convert model units

Model results are given in [mmol m-3], whereas GLODAP data are in [mu*mol kg-1]. This refers to the variables:

  • DIC
  • ALK
  • O2
  • NO3
  • PO4
  • SiO3

For comparison, model results were converted from [mmol m-3] to [mu*mol kg-1]

GLODAP_model <- GLODAP_model %>%
  mutate(rho = gsw_pot_rho_t_exact(
    SA = SALT_model,
    t = TEMP_model,
    p = depth,
    p_ref = 10.1325
  ))

GLODAP_model <- GLODAP_model %>%
  mutate(
    DIC_model  <- DIC_model * (1000 / rho),
    ALK_model  <- ALK_model * (1000 / rho),
    O2_model   <- O2_model * (1000 / rho),
    NO3_model  <- NO3_model * (1000 / rho),
    PO4_model  <- PO4_model * (1000 / rho),
    SiO3_model <- SiO3_model * (1000 / rho)
  )

3 Apply basin mask

GLODAP_model <- inner_join(GLODAP_model, basinmask)

4 Observation and model data comparison

# for loop across variables
obs_var <-
  c("tco2",
    "talk",
    "oxygen",
    "nitrate",
    "phosphate",
    "silicate",
    "sal",
    "tem")

model_var <-
  c(
    "DIC_model",
    "ALK_model",
    "O2_model",
    "NO3_model",
    "PO4_model",
    "SiO3_model",
    "SALT_model",
    "TEMP_model"
  )

# too many figures, here instead of length(obs_var), just use the first two.
for (i in 1:8) {
  # Uncomment line below for manual testing
  # i <- 2
  
  # select correlated observation and model variable
  GLODAP_model_var <- GLODAP_model %>%
    select(year,
           month,
           lat,
           lon,
           depth,!!sym(obs_var[i]),!!sym(model_var[i])) %>%
    drop_na() %>%
    mutate(
      season = case_when(
        month %in% c(3, 4, 5) ~ "Spring",
        month %in% c(6, 7, 8) ~ "Summer",
        month %in% c(9, 10, 11) ~ "Autumn",
        month %in% c(12, 1, 2) ~ "Winter"
      )
    )
  
  # calculate equal axis limits and binwidth
  axis_lims <- GLODAP_model_var %>%
    summarise(max_value = max(c(max(!!sym(
      obs_var[i]
    )),
    max(
      !!sym(model_var[i])
    ))),
    min_value = min(c(min(!!sym(
      obs_var[i]
    )),
    min(
      !!sym(model_var[i])
    ))))
  
  binwidth_value <- (axis_lims$max_value - axis_lims$min_value) / 40
  axis_lims <- c(axis_lims$min_value, axis_lims$max_value)
  
  
  
  # obs-model plot (season)
  print(
    ggplot(GLODAP_model_var, aes(
      x = !!sym(obs_var[i]),
      y = !!sym(model_var[i])
    )) +
      geom_bin2d(binwidth = binwidth_value) +
      labs(title = "Observation (x) vs Model (y)",
           subtitle = "Seasonal comparison") +
      scale_fill_viridis_c(trans = "log") +
      geom_abline(slope = 1,
                  col = 'red') +
      coord_equal(xlim = axis_lims,
                  ylim = axis_lims) +
      facet_wrap(~ season)
  )
  
  # obs-model plot (year)
  print(
    ggplot(GLODAP_model_var, aes(
      x = !!sym(obs_var[i]),
      y = !!sym(model_var[i])
    )) +
      geom_bin2d(binwidth = binwidth_value) +
      labs(title = "Observation (x) vs Model (y)",
           subtitle = "All years") +
      scale_fill_viridis_c(trans = "log") +
      geom_abline(slope = 1, col = 'red') +
      coord_equal(xlim = axis_lims,
                  ylim = axis_lims)
  )
  
  
  # Calculate variable offset
  GLODAP_model_var <- GLODAP_model_var %>%
    mutate(offset = !!sym(model_var[i]) -!!sym(obs_var[i]))
  
  # Calculate annual mean offset
  GLODAP_model_var_year <- GLODAP_model_var %>%
    group_by(year) %>%
    summarise(offset = mean(offset)) %>%
    ungroup()
  
  # plot annual mean offset
  print(
    GLODAP_model_var_year %>%
      ggplot(aes(year, offset)) +
      geom_point() +
      geom_line() +
      labs(title = "Annual mean offset",
           subtitle = paste(sym(model_var[i]), "-", sym(obs_var[i])))
  )
  
  # spatial distribution of the model-observations offset for 4 depth intervals
  intervals <- c(0, 150, 500, 2000, 8000)
  for (j in 1:4) {
    # j <- 1
    GLODAP_model_var_grid <- GLODAP_model_var %>%
      filter(depth >= intervals[j],
             depth < intervals[j + 1]) %>%
      group_by(lat, lon) %>%
      summarise(offset = mean(offset)) %>%
      ungroup()
    
    # plot mean spatial distribution (model - obs)
    limit <- max(abs(GLODAP_model_var_grid$offset)) * c(-1, 1)
    print(
      ggplot() +
        geom_raster(data = GLODAP_model_var_grid, aes(lon, lat, fill = offset)) +
        scale_fill_scico(
          palette = "vik",
          name = "offset",
          limit = limit
        ) +
        coord_quickmap(expand = 0) +
        labs(
          title = paste(
            model_var[i],
            "vs",
            obs_var[i],
            "mean offset spatial distribution"
          ),
          subtitle = paste("Depth interval:",
                           intervals[j], "-", intervals[j + 1]),
          x = "Longitude",
          y = "Latitude"
        ) +
        geom_raster(data = landmask,
                    aes(lon, lat), fill = "grey80")
    )
    
    print(GLODAP_model_var_grid %>%
            ggplot(aes(offset)) +
            geom_histogram()) +
      labs(
        title = paste(model_var[i],
                      "vs",
                      obs_var[i],
                      "mean offset per grid cell - histogram"),
        subtitle = paste("Depth interval:",
                         intervals[j], "-", intervals[j + 1]))
  }
}
